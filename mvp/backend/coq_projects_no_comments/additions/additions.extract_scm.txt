Extraction Language Scheme.

Axiom int : Set. 
Axiom i2n : int -> nat. 
Axiom z2i : Z -> int. 

Extract Inlined Constant int => "int". 

Extract Constant i2n =>
 "(lambda (i) (if (zero? i) `(O) `(S ,(i2n (- i 1)))))".

Extract Constant z2i =>
 "(lambda (z)
    (define (p2i p)
      (match p
        ((XH) 1)
        ((XO p) (* 2 (p2i p)))
        ((XI p) (+ 1 (* 2 (p2i p))))))
    (match z
       ((Z0) 0)
       ((Zpos p) (p2i p))
       ((Zneg p) (- (p2i p)))))
  ".

Extraction Inline Wf_nat.gt_wf_rec Wf_nat.lt_wf_rec.
