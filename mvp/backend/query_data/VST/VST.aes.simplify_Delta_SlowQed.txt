  let i := Zlength w in
  let temp := (Znth (i-1) w) in
  let temp' := if (i mod Nk =? 0) then
    Int.xor (SubWord (RotWord temp)) (Znth (i/Nk) RCon)

    SubWord temp

    temp
  in
    w ++ [Int.xor (Znth (i-8) w) temp'].
