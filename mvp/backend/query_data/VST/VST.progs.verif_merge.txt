     v p.

Fixpoint merge l1 :=

  match l1 with

  | nil => (fun l2 => l2)

  | h1 :: t1 =>

      let fix merge_aux l2 :=

        match l2 with

        | nil => l1

        | h2 :: t2 =>

           if Int.cmp Cle h1 h2

           then h1 :: merge t1 l2

           else h2 :: merge_aux t2

        end

      in merge_aux

  end.
