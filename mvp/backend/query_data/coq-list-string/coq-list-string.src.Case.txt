Import ListNotations.

Import LString.



Definition capitalize (s : t) : t :=

  match s with

  | [] => []

  | c :: s => Char.up_case c :: s

  end.



Definition down_case (s : t) : t :=

  List.map Char.down_case s.



Definition up_case (s : t) : t :=

  List.map Char.up_case s.
