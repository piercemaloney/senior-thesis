Definition In_list (N : Set) (n : N) (Y : list N) : Prop :=
  (fix F (l : list N) : Prop :=
     match l with
     | nil => False
     | cons y l0 => n = y :>N \/ F l0
     end) Y.

Definition In_ehlist (a : act) (H : ehlist) : Prop :=
  (fix F (e : ehlist) : Prop :=
     match e with
     | ehnil => False
     | ehcons a0 e0 => a = a0 :>act \/ F e0
     end) H.
